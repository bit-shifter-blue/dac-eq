From the Terminal:

  # 1. Clone the repository                                                                                                                                  
  git clone git@github.com:bit-shifter-blue/dac-eq.git
  cd dac-eq

  # 2. Run install script
  ./install.sh

  # 3. Activate virtual environment
  source venv/bin/activate

  # 4. Start Claude Code
  claude
